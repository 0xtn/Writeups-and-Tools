	$pathe2= "$root/index.html";
	$pathe3= "$root/index.htm";
	if (file_exists($pathe1)){
		$pathe_index = fopen($pathe1,"w+");
		$put_index = fwrite($pathe_index,$index);
		if (!$put_index) {
			unlink("$pathe1");
			$put_index = fwrite($pathe_index,$index);
		}
	}
	if (file_exists($pathe2)){
		$pathe_index = fopen($pathe2,"w+");

			unlink("$pathe2");

		}
	}

		$pathe_index = fopen($pathe3,"w+");
		$put_index = fwrite($pathe_index,$index);
		if (!$put_index) {
			unlink("$pathe3");
			$put_index = fwrite($pathe_index,$index);
		}
	}
